  The agency rebuilds this file mechanically each RECONCILE. Below is the

    zone1_overview.scope.out[]     <- mission.md "Scope (out)" bullets.

                                       (Sponsor=CEO/orchestrator; Owner=orchestrator; divisions listed).
    zone1_overview.timeline.start  <- mission.md "_Last updated_" date of the kickoff run (or run start).
    zone1_overview.timeline.target <- mission.md target/deadline if stated, else "TBD".
    zone1_overview.timeline.status_label <- derived from zone3 (In progress / Done / Blocked).

  ----------------------------------------------------------------------
  ZONE 2 — zone2_arc        <-  state/task-graph.md  (phases / batches)
  ----------------------------------------------------------------------
    zone2_arc.unit           <- "phase" (the task-graph groups tasks under P1/P2/P3 + Quality/Mgmt).
                                 Use "batch" instead if rendering the topological-batch view.
    zone2_arc.items[].id     <- the phase id (P1, P2, P3, Q, M) — or batch number.
    zone2_arc.items[].name   <- the phase heading in task-graph.md ("Dashboard artifact", etc.).
    zone2_arc.items[].one_line <- a one-line summary of that phase's tasks.
    zone2_arc.items[].status <- ROLL-UP of the phase's task statuses in task-graph.md:
                                   all tasks done           -> "done"
                                   any task in_progress, OR
                                     some done + some pending -> "in_progress"
                                   all tasks pending/blocked  -> "upcoming"

  ----------------------------------------------------------------------
  ZONE 3 — zone3_you_are_here  <-  state/task-graph.md  (active phase)
  ----------------------------------------------------------------------
    active_phase = earliest phase (by arc order) containing an in_progress
                   or pending task in task-graph.md.
    zone3.active_phase_id    <- that phase's id.
    zone3.active_phase_name  <- that phase's name.
    zone3.position_index     <- 1-based index of active phase in zone2_arc.items.
    zone3.position_total     <- count of zone2_arc.items.
    zone3.position_label     <- "Phase {position_index} of {position_total}".
    zone3.arc_percent        <- round( (number of arc items with status==done
                                  + 0.5 * arc items in_progress) / position_total * 100 ).
    zone3.delivers           <- the active phase's "delivers" note (from its task-graph notes /
                                 phase objective).
    zone3.why_it_matters     <- the active phase's rationale (from task-graph notes / mission).
